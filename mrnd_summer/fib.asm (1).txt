
DATA NUM
DATA RES[15]
CONST M=0
CONST N=1
CONST COUN=0
CONST ZERO=0
CONST ONE=1
READ NUM
MOV CX,NUM
MOV BX,N
MOV RES[COUN],M
MOV DX,COUN
MOV EX,ZERO
MOV FX,ONE
ADD DX,DX,EX
MOV COUN,DX
X:
IF BX LT CX THEN
MOV AX,M
MOV RES[COUN],N
MOV DX,COUN
ADD DX,DX,FX
MOV COUN,DX
ADD N,AX,BX
MOV M,BX
MOV BX,N
JMP X
;
END
